% Channel: lida_weather_log_1min
readAPIKey = 'XXXXXXXXXXXXXXXX';
readChannelID = 1306405;
TemperatureFieldID = 1;
HumidityFieldID = 2;
Days = 14;

data = thingSpeakRead(readChannelID,'Fields',[TemperatureFieldID HumidityFieldID], ...
                                               'NumDays', Days, ...
                                               'ReadKey',readAPIKey);

temperatureData = data(:,1);
humidityData = data(:,2);

% Visualize the data
figure
binscatter(temperatureData,humidityData);
xlabel('Temperature');
ylabel('Humidity');
title(strcat('Плотность температуры и влажности за ', num2str(Days, ' %d'), ' дней'));
